# Basic Tour

The simplest possible tour setup — just steps and the overlay.

## Tour Config

```typescript
// tour.config.ts
import type { TourConfig } from 'nuxt-guided-tour/dist/runtime/types'

export const tourConfig: TourConfig = {
  version: 1,
  autoStart: true,
  autoStartDelay: 500,
  storageKeyPrefix: 'basic-tour',
  steps: [
    {
      id: 'nav',
      target: '[data-tour="nav"]',
      title: 'Navigation',
      content: 'Use the nav bar to move between pages.',
      icon: 'i-heroicons-bars-3',
      position: 'bottom'
    },
    {
      id: 'search',
      target: '[data-tour="search"]',
      title: 'Search',
      content: 'Search for anything using the search bar.',
      icon: 'i-heroicons-magnifying-glass',
      position: 'bottom',
      shortcuts: [
        { label: 'Focus Search', mac: ['⌘', 'K'], win: ['Ctrl', 'K'] }
      ]
    },
    {
      id: 'profile',
      target: '[data-tour="profile"]',
      title: 'Your Profile',
      content: 'Access settings and preferences from your profile.',
      icon: 'i-heroicons-user-circle',
      position: 'left'
    }
  ]
}
```

## Page Component

```vue
<script setup>
import { tourConfig } from '~/tour.config'

const tour = useTour(tourConfig)

onMounted(() => {
  if (tour.autoStart && !tour.hasCompletedTour.value) {
    setTimeout(() => tour.start(), tour.autoStartDelay)
  }
})
</script>

<template>
  <nav data-tour="nav">Navigation Bar</nav>
  <input data-tour="search" placeholder="Search..." />
  <div data-tour="profile">Profile</div>

  <TourOverlay
    :is-active="tour.isActive.value"
    :current-step="tour.currentStep.value"
    :progress="tour.progress.value"
    @next="tour.next"
    @previous="tour.previous"
    @cancel="tour.cancel"
  />
</template>
```

This is the minimal setup. No welcome modal, no intro slides — the tour starts immediately.
