# Types

All TypeScript types exported by `nuxt-guided-tour`.

## TourPosition

```typescript
type TourPosition = 'top' | 'bottom' | 'left' | 'right'
```

## TourAlign

```typescript
type TourAlign = 'start' | 'center' | 'end'
```

## TourStep

```typescript
interface TourStep {
  id: string
  target: string
  title: string
  content: string
  tip?: string
  position?: TourPosition
  align?: TourAlign
  highlight?: boolean
  icon?: string
  shortcut?: string[]       // deprecated
  shortcutWin?: string[]    // deprecated
  shortcuts?: Array<{
    label: string
    mac: string[]
    win: string[]
  }>
}
```

## TourConfig

```typescript
interface TourConfig {
  version: number
  autoStart: boolean
  autoStartDelay: number
  storageKeyPrefix: string
  steps: TourStep[]
  announce?: (message: string, priority?: 'polite' | 'assertive') => void
}
```

## TourProgress

```typescript
interface TourProgress {
  current: number    // 1-indexed
  total: number
  percentage: number // 0-100
}
```

## IntroSlide

```typescript
interface IntroSlide {
  title: string
  icon: string
  content: string[]
  examples?: IntroSlideExample[]
  benefits?: IntroSlideBenefit[]
  conversions?: IntroSlideConversion[]
  callout?: IntroSlideCallout
  footer?: string
}
```

## IntroSlideExample

```typescript
interface IntroSlideExample {

```

## IntroSlideBenefit

interface IntroSlideBenefit {

## UseTourReturn

```typescript
interface UseTourReturn {
  steps: TourStep[]
  version: number
  autoStart: boolean
  autoStartDelay: number
  currentStepIndex: Ref<number>
  currentStep: ComputedRef<TourStep | null>
  isActive: ComputedRef<boolean>
  hasCompletedTour: Ref<boolean>
  progress: ComputedRef<TourProgress>
  start: () => void
  next: () => void
  previous: () => void
  cancel: () => void
  complete: () => void
  goToStep: (index: number) => void
  getStepById: (id: string) => TourStep | undefined
  getStepIndex: (id: string) => number
  handleKeydown: (event: KeyboardEvent) => void
  markAsSeen: () => void
  resetCompletion: () => void
}
```
